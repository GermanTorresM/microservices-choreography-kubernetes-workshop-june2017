Oracle Java 8 JRE from Docker Store
image:  store/oracle/serverjre:8

For Maven in Docker - see https://github.com/driv/docker-maven-java-oracle/tree/master/jdk-8
and https://codefresh.io/blog/java_docker_pipeline/

docker build -t lucasjellema/java-app-runner .


docker run -e "GIT_URL=https://github.com/lucasjellema/java-express-style-rest-api" -e "APP_PORT=8080" -p 8001:8765 -e "APP_HOME=." -e "ARTIFACT_ID=my-rest" -e "VERSION=1.0-SNAPSHOT"  -e "APP_STARTUP=nl.amis.rest.App2" -e "MAVEN_DEPENDENCIES=LIBS"   lucasjellema/java-app-runner
